// ════════════════════════════════════════════
//  MAGNETIC FIELD PARTICLE SYSTEM  v3
//  ─ Horizontal dipole axis → field lines loop VERTICALLY
//  ─ Dipole poles are LEFT and RIGHT of center
//  ─ Particles travel big vertical ovals, passing through center = FILL
//  ─ Big loop = long hold, small loop = fast scalp
//  ─ Direction: from top → right-arc → bottom → left-arc → top (CW when viewed straight)
//    Reversed 180°: from top → left-arc → bottom → right-arc → top (CCW)
// ════════════════════════════════════════════
const MAG = (() => {
  const c = document.getElementById('magC');
  const ctx = c.getContext('2d');
  let W, H, cx, cy;
  let fillCount = 0;
  const fills = [];

  // ── Horizontal-axis dipole field line parametric ──
  // Dipole axis = X axis (left↔right poles)
  // A field line is parameterized by its "opening" angle α from the axis
  // Parametric: for angle α (0=tight near axis, π/2=widest equatorial loop)
  //   r(λ) = R·cos²(λ)   where λ sweeps -π..π (latitude from equatorial plane)
  //   x = cx + r·cos(λ)·sin(α_side)   ← horizontal spread stays narrow
  //   y = cy + r·sin(λ)                ← vertical excursion is large
  //
  // This gives tall vertical ovals that pinch at cx,cy (center/equator)
  // and bulge up/down — exactly like magnetic field lines around a bar magnet
  // lying on its side.

  function dipoleXY(lambda, R, side) {
    // lambda: -π → π sweeps one full loop
    // R: max radius of this field line
    // side: +1 right-upper loop, -1 left-lower loop (two lobes)
    const r = R * Math.cos(lambda) * Math.cos(lambda);
    // With horizontal axis: poles on X, loops go vertical
    const px = cx + r * Math.cos(lambda) * side * 0.45; // slight horizontal sway
    const py = cy - r * Math.sin(lambda);               // vertical = main axis of loop
    return { x: px, y: py };
  }

  class Particle {
    constructor() { this.reset(); }

    reset() {
      // Which lobe: right (+1) or left (-1)
      this.side = Math.random() < 0.5 ? 1 : -1;
      // Loop size: big=long hold, small=scalp
      this.loopScale = 0.25 + Math.random() * 0.75;
      // Max radius scaled to canvas — use more vertical space
      this.R = (H * 0.44) * this.loopScale;
      // Speed: slower overall, big loops even slower
      // Reversed: subtract phi (CCW, reversed 180°)
      this.speed = (0.004 + Math.random() * 0.005) / Math.max(0.4, this.loopScale);
      // Start at random point on its loop
      this.phi = (Math.random() - 0.5) * Math.PI * 2;
      this.birthTime = Date.now();
      this.fillTriggered = false;
      this.pnl = (Math.random() * 70 - 12).toFixed(1);
      this.profit = parseFloat(this.pnl) >= 0;
      this.alpha = 0.45 + Math.random() * 0.45;
      this.size = 1.8 + this.loopScale * 2.0;
      this.trail = [];
      this.trailLen = Math.floor(12 + this.loopScale * 35);
      this.dead = false;
      this.lifetime = 6000 + this.loopScale * 14000;
    }

    update() {
      // Reversed direction: phi decreases (CCW = 180° flip vs previous CW)
      this.phi -= this.speed;
      if (this.phi < -Math.PI) this.phi += Math.PI * 2;

      const pos = dipoleXY(this.phi, this.R, this.side);
      this.trail.push({ x: pos.x, y: pos.y });
      if (this.trail.length > this.trailLen) this.trail.shift();
      this.x = pos.x;
      this.y = pos.y;

      // Fill event: particle passes near center (equator crossing)
      const dist = Math.hypot(this.x - cx, this.y - cy);
      if (dist < 12 && !this.fillTriggered) {
        this.fillTriggered = true;
        fillCount++;
        Backend.fillCount = fillCount;
        fills.push({ x: cx, y: cy, t: 1.0, pnl: this.pnl, profit: this.profit });
        setTimeout(() => { this.fillTriggered = false; }, 1200);
        const now = new Date();
        const ts = `${String(now.getHours()).padStart(2,'0')}:${String(now.getMinutes()).padStart(2,'0')}:${String(now.getSeconds()).padStart(2,'0')}`;
        DataC.logLines.push([ts, this.profit ? 'EXEC' : 'LOSS', `fill ${this.profit?'+':''}$${this.pnl}`]);
        if (DataC.logLines.length > 40) DataC.logLines.shift();
        renderLog();
      }

      if (Date.now() - this.birthTime > this.lifetime) this.dead = true;
    }

    draw() {
      // Trail
      const tlen = this.trail.length;
      for (let i = 1; i < tlen; i++) {
        const p = this.trail[i];
        const ratio = i / tlen;
        const r = this.profit ? `rgba(77,255,210,` : `rgba(255,80,80,`;
        ctx.beginPath();
        ctx.arc(p.x, p.y, 0.7 + ratio * 0.8, 0, Math.PI * 2);
        ctx.fillStyle = r + (ratio * this.alpha * 0.55) + ')';
        ctx.fill();
      }
      // Head
      ctx.beginPath();
      ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2);
      ctx.fillStyle = this.profit ? '#4dffd2' : '#ff5050';
      ctx.globalAlpha = this.alpha;
      ctx.fill();
      ctx.globalAlpha = 1;
    }
  }

  const particles = [];
  let orbT = 0;
  let bgFieldLines = [];

  function buildFieldLines() {
    bgFieldLines = [];
    // Draw background field lines (static, horizontal dipole = vertical ovals)
    const sides = [1, -1];
    const scales = [0.18, 0.32, 0.48, 0.64, 0.80, 0.95];
    sides.forEach(side => {
      scales.forEach((sc, si) => {
        const R = H * 0.44 * sc;
        const pts = [];
        for (let t = -Math.PI; t <= Math.PI; t += 0.04) {
          pts.push(dipoleXY(t, R, side));
        }
        bgFieldLines.push({ pts, alpha: 0.018 + si * 0.008 });
      });
    });
  }

  function init() {
    W = c.parentElement.clientWidth;
    H = 340;
    c.width = W;
    c.height = H;
    cx = W / 2;
    cy = H / 2;
    buildFieldLines();
    // Stagger initial spawn
    for (let i = 0; i < 14; i++) setTimeout(() => spawnParticle(), i * 250);
  }

  function spawnParticle() {
    if (particles.filter(p => !p.dead).length < 22) particles.push(new Particle());
  }

  function drawFieldLines() {
    bgFieldLines.forEach(({ pts, alpha }) => {
      ctx.beginPath();
      pts.forEach((p, i) => i === 0 ? ctx.moveTo(p.x, p.y) : ctx.lineTo(p.x, p.y));
      ctx.strokeStyle = `rgba(0,90,65,${alpha})`;
      ctx.lineWidth = 0.5;
      ctx.stroke();
    });
    // Pole markers (left & right)
    [cx - W*0.04, cx + W*0.04].forEach((px, i) => {
      ctx.beginPath(); ctx.arc(px, cy, 3, 0, Math.PI * 2);
      ctx.fillStyle = 'rgba(77,255,210,0.12)'; ctx.fill();
      ctx.font = '7px Courier New';
      ctx.fillStyle = 'rgba(77,255,210,0.18)';
      ctx.fillText(i === 0 ? 'S' : 'N', px - 3, cy + 12);
    });
  }

  function drawCenterSphere() {
    const pulse = 0.75 + 0.25 * Math.sin(orbT * 2.5);
    // Outer halo rings
    [28, 22, 16].forEach((r, i) => {
      ctx.beginPath();
      ctx.arc(cx, cy, r * pulse, 0, Math.PI * 2);
      ctx.strokeStyle = `rgba(77,255,210,${0.03 + i * 0.02})`;
      ctx.lineWidth = 1;
      ctx.stroke();
    });
    // Main sphere ring
    ctx.beginPath();
    ctx.arc(cx, cy, 7, 0, Math.PI * 2);
    ctx.strokeStyle = `rgba(77,255,210,${0.55 * pulse})`;
    ctx.lineWidth = 1.2;
    ctx.stroke();
    // Inner fill
    ctx.beginPath();
    ctx.arc(cx, cy, 4, 0, Math.PI * 2);
    ctx.fillStyle = `rgba(77,255,210,${0.55 * pulse})`;
    ctx.fill();
    // Crosshair
    ctx.strokeStyle = 'rgba(77,255,210,0.12)';
    ctx.lineWidth = 0.5;
    ctx.beginPath(); ctx.moveTo(cx - 28, cy); ctx.lineTo(cx + 28, cy); ctx.stroke();
    ctx.beginPath(); ctx.moveTo(cx, cy - 28); ctx.lineTo(cx, cy + 28); ctx.stroke();
  }

  function drawFills() {
    for (let i = fills.length - 1; i >= 0; i--) {
      const f = fills[i];
      f.t -= 0.012;
      if (f.t <= 0) { fills.splice(i, 1); continue; }
      // Expanding ripple
      const r = (1 - f.t) * 55;
      ctx.beginPath();
      ctx.arc(f.x, f.y, r, 0, Math.PI * 2);
      ctx.strokeStyle = f.profit ? `rgba(77,255,210,${f.t * 0.6})` : `rgba(255,80,80,${f.t * 0.6})`;
      ctx.lineWidth = 1;
      ctx.stroke();
      // Floating PnL label
      const ty = f.y - (1 - f.t) * 38;
      ctx.font = '8px Courier New';
      ctx.fillStyle = f.profit ? `rgba(77,255,210,${f.t})` : `rgba(255,80,80,${f.t})`;
      ctx.fillText((f.profit ? '+' : '') + '$' + f.pnl, f.x + 10, ty);
    }
  }

  function frame() {
    ctx.clearRect(0, 0, W, H);
    orbT += 0.008;

    drawFieldLines();

    for (let i = particles.length - 1; i >= 0; i--) {
      if (particles[i].dead) particles.splice(i, 1);
    }
    if (Math.random() < 0.022) spawnParticle();

    particles.forEach(p => { p.update(); p.draw(); });

    drawCenterSphere();
    drawFills();

    const active = particles.filter(p => !p.dead).length;
    Backend.activeParticles = active;
    document.getElementById('activeCount').textContent = active;
    document.getElementById('fillCount').textContent = fillCount;
    const avgHold = active > 0
      ? (particles.reduce((a, p) => a + (Date.now() - p.birthTime), 0) / active / 1000).toFixed(1)
      : '0.0';
    document.getElementById('avgHold').textContent = avgHold + 's';
    document.getElementById('throughput').textContent = Math.floor(active * 2.8) + '/min';

    requestAnimationFrame(frame);
  }

  return { init, frame };
})();

// ════════════════════════════════════════════
//  BAYESIAN CANVAS  (reads DataA.bayesMu)
// ════════════════════════════════════════════
function drawBayes() {
  const c = document.getElementById('bayesC');
  const W = c.parentElement.clientWidth - 16;
  c.width = W * 2; c.height = 140;
  const ctx = c.getContext('2d');
  ctx.scale(2, 2);
  const cW = W, cH = 70;
  ctx.clearRect(0, 0, cW, cH);
  function gauss(x, mu, sig) { return Math.exp(-0.5*((x-mu)/sig)**2)/(sig*Math.sqrt(2*Math.PI)); }
  const curves = [
    { mu: 0.38, sig: 0.13, col: '#333', label: 'Prior' },
    { mu: DataA.bayesMu, sig: 0.09, col: '#4dffd2', label: 'Posterior' },
    { mu: Math.min(DataA.bayesMu + 0.07, 0.80), sig: 0.065, col: '#2a7a5a', label: 'MAP' },
  ];
  curves.forEach(({ mu, sig, col }) => {
    ctx.beginPath();
    for (let i = 0; i <= cW; i++) {
      const x = i / cW, y = gauss(x, mu, sig);
      const py = cH - y * cH * 2.6;
      i === 0 ? ctx.moveTo(i, py) : ctx.lineTo(i, py);
    }
    ctx.strokeStyle = col; ctx.lineWidth = 1; ctx.stroke();
  });
  ctx.font = '7px Courier New';
  curves.forEach(({ col, label }, i) => { ctx.fillStyle = col; ctx.fillText(label, 4, 10 + i * 10); });
  // Vertical posterior mean line
  ctx.strokeStyle = 'rgba(77,255,210,0.2)'; ctx.lineWidth = 0.5; ctx.setLineDash([2,2]);
  ctx.beginPath(); ctx.moveTo(DataA.bayesMu * cW, 0); ctx.lineTo(DataA.bayesMu * cW, cH); ctx.stroke();
  ctx.setLineDash([]);
}

// ════════════════════════════════════════════
//  SPREAD CANVAS  (reads DataA.spreadOffset)
// ════════════════════════════════════════════
function drawSpread() {
  const c = document.getElementById('spreadC');
  const W = c.parentElement.clientWidth - 16;
  c.width = W * 2; c.height = 140;
  const ctx = c.getContext('2d');
  ctx.scale(2, 2);
  const cW = W, cH = 70;
  ctx.clearRect(0, 0, cW, cH);
  // Mid line
  ctx.strokeStyle = '#1a1a1a'; ctx.lineWidth = 0.5;
  ctx.beginPath(); ctx.moveTo(0, cH/2); ctx.lineTo(cW, cH/2); ctx.stroke();
  // Threshold lines ±2σ
  const threshY = cH/2 - cH * 0.35;
  const threshY2 = cH/2 + cH * 0.35;
  ctx.strokeStyle = '#1a2a1a'; ctx.setLineDash([2,3]);
  ctx.beginPath(); ctx.moveTo(0, threshY); ctx.lineTo(cW, threshY); ctx.stroke();
  ctx.beginPath(); ctx.moveTo(0, threshY2); ctx.lineTo(cW, threshY2); ctx.stroke();
  ctx.setLineDash([]);
  // Spread line
  ctx.beginPath();
  for (let i = 0; i <= cW; i++) {
    const t = (i + DataA.spreadOffset) * 0.07;
    const z = Math.sin(t)*1.4 + Math.sin(t*2.1)*0.6 + Math.sin(t*0.3)*0.4;
    const y = cH/2 - z * (cH * 0.28);
    i === 0 ? ctx.moveTo(i, y) : ctx.lineTo(i, y);
    // Arb signal when |z| > 2
    if (Math.abs(z) > 1.6 && i % 18 < 12) {
      ctx.fillStyle = z > 0 ? 'rgba(77,255,210,0.15)' : 'rgba(255,80,80,0.15)';
      ctx.fillRect(i - 1, cH/2 - Math.abs(z)*(cH*0.28), 2, Math.abs(z)*(cH*0.28));
    }
  }
  ctx.strokeStyle = '#2a7a5a'; ctx.lineWidth = 1; ctx.stroke();
  // Z label
  DataA.spreadHistory.push(Backend.spreadZ);
  if (DataA.spreadHistory.length > 200) DataA.spreadHistory.shift();
  const curZ = Backend.spreadZ.toFixed(2);
  ctx.font = '6px Courier New'; ctx.fillStyle = '#444';
  ctx.fillText('z=' + curZ, cW - 30, 8);
  ctx.fillText('±2σ', 2, threshY - 1);
}

  for (let i = 0; i < 200; i++) { v += Math.random()*2.5-0.2; if(v<0)v=0; DataC.pnlPts.push(v); }

function drawPnL() {
  const c = document.getElementById('pnlC');
  const W = c.parentElement.clientWidth - 16;
  c.width = W * 2; c.height = 150;
  const ctx = c.getContext('2d');
  ctx.scale(2, 2);
  const cW = W, cH = 75;
  ctx.clearRect(0, 0, cW, cH);
  const pts = DataC.pnlPts.slice(-cW);
  const mx = Math.max(...pts, 1);
  ctx.beginPath();
  pts.forEach((p, i) => {
    const x = i, y = cH - (p / mx) * cH * 0.88;
    i === 0 ? ctx.moveTo(x, y) : ctx.lineTo(x, y);
  });
  ctx.strokeStyle = '#fff'; ctx.lineWidth = 1; ctx.stroke();
  // Current dot
  if (pts.length > 1) {
    const lx = pts.length - 1, ly = cH - (pts[pts.length-1]/mx)*cH*0.88;
    ctx.beginPath(); ctx.arc(lx, ly, 2, 0, Math.PI*2);
    ctx.fillStyle = '#4dffd2'; ctx.fill();
  }
}

// ════════════════════════════════════════════
function masterTick() {
  Backend.tick();
  DataC.tickCount++;

  // PnL
  DataC.pnlVal += Math.round(Math.random() * 130 - 20);
  Backend.totalPnL = DataC.pnlVal;
  document.getElementById('pnlNum').textContent = '$' + DataC.pnlVal.toLocaleString();
  document.getElementById('magPnl').textContent = '+$' + (Math.random()*60+10).toFixed(1);
  DataC.pnlPts.push(DataC.pnlPts[DataC.pnlPts.length-1] + Math.random()*2.2 - 0.18);
  drawPnL();

  // Spread + Bayes
  DataA.spreadOffset += 4;
  drawSpread();
  DataA.bayesMu = Backend.bayesPosterior;
  if (DataC.tickCount % 3 === 0) drawBayes();

  // Kelly
  renderKelly();

  // Factors
  updateFactors();
  renderSignals();
  renderStatus();
  updateApiPanel();
  buildTicker();

  // Block num
  const bn = document.getElementById('blockNum');
  const cur = parseInt(bn.textContent.replace(/,/g,''));
  bn.textContent = (cur + 1).toLocaleString();

  // Random log events
  if (Math.random() < 0.4) {
    const tags = ['ARBS','BAYES','SPREAD','EXEC','SCAN','SIGNAL','KELLY','OKX','FILL'];
    const now = new Date();
    const ts = `${String(now.getHours()).padStart(2,'0')}:${String(now.getMinutes()).padStart(2,'0')}:${String(now.getSeconds()).padStart(2,'0')}`;
    const vals = [
      'z=' + Backend.spreadZ.toFixed(2),
      'p=' + Backend.bayesPosterior.toFixed(3),
      'f*=' + Backend.kellyF.toFixed(3),
      'BTC $' + Backend.btcPrice.toFixed(0),
      '+$' + (Math.random()*50).toFixed(1),
      'lag=' + Backend.latency.toFixed(1) + 'ms',
      'opp detected',
      'sizing ' + Backend.kellyF.toFixed(3),
    ];
    DataC.logLines.push([ts, tags[Math.random()*tags.length|0], vals[Math.random()*vals.length|0]]);
    if (DataC.logLines.length > 40) DataC.logLines.shift();
    renderLog();
  }

  // Drawdown update
  const dd = (Math.random()*1.5).toFixed(1);
  document.getElementById('ddVal').textContent = dd + '%';
  document.getElementById('ddVal').className = parseFloat(dd) > 1.5 ? 'sr' : 'sg';
}
